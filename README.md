## Configuration properties
```yml
sbuf:
  aspect:
    base-package: 
    controller:
      enabled:
      traced-headers:
      success-operation-trace.enable:
    service:
      enabled:
    repository:
      enabled:
    logging-component:
      enabled:
  config:
    mongo:
      enabled:
    tracing:
      on-database:
  security:
    basic:
      client-id:
      client-secret:
  service:
    smart-cache:
      enabled:
```
